% Params: width (default: 0.75\textwidth), filename in the images/ folder, caption, fig reference (e.g., ~\ref{fig:funnel})
\newcommand{\addFigure}[4][0.2] {  % Add single figure
\begin{figure}[!ht]
    \vspace{0.45cm}
    \centering

    \includegraphics[width=0.45\textwidth, trim={0 0 0 #1cm}, clip]{images/#2}
    \caption{\roboto\selectfont\textcolor{Gray}{\uppercase{#3}}}
    \label{fig:#4}

\end{figure}
}

% Params: image filename, caption, fig reference, place at t (top) or b (bottom) of page, subcaptions
\usepackage{stfloats}
\newcommand{\addFigureSet}[8] {  % Add 2x2 figure set
\begin{figure*}[#4]
    \vspace{0.45cm}
    \centering

    \begin{subfigure}[b]{.45\linewidth}
      \includegraphics[width=\linewidth, cfbox=Gray 0.4pt]{images/maps/#1_1}
      \caption{#5}
    \end{subfigure}\hfill
    \begin{subfigure}[b]{.45\linewidth}
      \includegraphics[width=\linewidth, cfbox=Gray 0.4pt]{images/maps/#1_2}
      \caption{#6}
    \end{subfigure}

    \smallskip

    \begin{subfigure}[b]{.45\linewidth}
      \includegraphics[width=\linewidth, cfbox=Gray 0.4pt]{images/maps/#1_3}
      \caption{#7}
    \end{subfigure}\hfill
    \begin{subfigure}[b]{.45\linewidth}
      \includegraphics[width=\linewidth, cfbox=Gray 0.4pt]{images/maps/#1_4}
      \caption{#8}
    \end{subfigure}

    \caption{\roboto\selectfont\textcolor{Gray}{\uppercase{#2}}}
    \label{fig:#3}

    \ifnum\pdfstrcmp{#4}{b}=0
      \vspace{-1cm}
    \else
      \vspace{0.45cm}
    \fi

\end{figure*}
}


% Tables ~~~~

\usepackage[figuresright]{rotating}
\usepackage{multirow}

% Params: line spacing (default: 10pt), caption, tbl reference (or `x` if continued table)
\newenvironment{lscape-env}[3][10]{  % landscape environment
    \clearpage
    \newgeometry{top=0.5in, bottom=0.5in, inner=0in, outer=0in}
    \thispagestyle{lscape-style}
    \begin{landscape}
    \begin{table*}[t]
    \vspace{-0.5cm}
    \ifnum\pdfstrcmp{#3}{x}=0
      \ContinuedFloat
    \fi
    \ifnum\pdfstrcmp{#2}{x}=-1
      \caption{\roboto\selectfont\textcolor{Gray}{\uppercase{#2}}}\label{tbl:#3}
    \fi
    \fontsize{8pt}{#1pt}\selectfont
}
{
    \end{table*}
    \end{landscape}
    \clearpage
}

% Params: top or bottom of page (default: b), filename in the tables/ folder, caption, tbl reference
\newcommand{\addTable}[4][b] {  % Add Table

\begin{table*}[#1]

    \centering

    \begingroup
      \fontsize{8pt}{10pt}\selectfont
      \input{tables/#2.tex}
    \endgroup

    \vspace{0.2cm}
    \caption{\roboto\selectfont\textcolor{Gray}{\uppercase{#3}}}
    \label{tbl:#4}

    \ifnum\pdfstrcmp{#1}{b}=0
      \vspace{-1cm}
    \fi

\end{table*}

}
